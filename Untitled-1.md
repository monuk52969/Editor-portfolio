gsap.registerPlugin(ScrollTrigger);

const videos = document.querySelectorAll("video");

videos.forEach((video, index) => {
  gsap.to(video, {
    opacity: 0,
    scrollTrigger: {
      trigger: video,
      start: () => "top+=" + (window.innerHeight * index) + " top",
      end: () => "top+=" + (window.innerHeight * (index + 1)) + " top",
      scrub: 2,
    }
  });
});
